      <!-- 展示当前头像 -->
      <div id = 'photo_img'> 
          <center>
          <Avatar :src="photo" :width='width' fit='fill'></Avatar>
          </center>

          <!-- 前端上传头像 -->
          <input type="file" @change='change($event)'>
      </form>
  </template>

  <script>

  import { config } from '../config';

  export default {

      data:(){
          photo:''
      }

      methods:{
        
          change:function(event){
                
              console.log(event.target.files)

              var data = new FormData()

              data.append('file',event.target.files[0])
                
              this.axios.post('/api/upload_file/',data,{headers: {'Content-Type': 'multipart/form-data'}}).then(resp=>{
                    
                  this.photo = config.baseurl + '/static/upload/' +resp.data.data

              })
          },
      }
  }

  </script>

      # 上传文件
  class UploadFile(APIView):
      def post(self, request):
          # 定义相应对象
          res = {}
          # 后端获取文件名，以uuid重命名，并保存到本地
          myFile = request.FILES.get('file')
          file_name = str(uuid.uuid4()) + myFile.name[-4:]
          f = open(os.path.join(UPLOAD_ROOT, '', file_name), 'wb')
          for chunk in myFile.chunks():
              f.write(chunk)
          # 关闭文件流
          f.close()

          # 返回新头像名
          res['data'] = file_name

          return Response(res)
