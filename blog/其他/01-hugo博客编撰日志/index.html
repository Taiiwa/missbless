自动生成目录

'''
这个脚本用来实现自动生成分级目录，项目打包，自动替换打包后的简历文件，配置代理并上传到github
'''

import os

dir_list = os.listdir('./content/blog/')

dir_list.remove('其他')
dir_list.append('其他')

from datetime import datetime

# 写入头部
date = "%sT%s"%(datetime.now().date(), datetime.now().time())
date = date.split('.',-1)[0] + '+08:00'

head = "---\ntitle: 'Contents'\ndate: %s\n---\n"%date

with open('./content/Contents.md', 'w', encoding='utf-8') as f:
    f.write(head)

# 生成博客目录
with open('./content/Contents.md', 'a', encoding='utf-8') as f:
    # 遍历每个文件夹
    for dir in dir_list:
        # 写入文件夹名：
        f.write('## ' + dir + '\n')
        # 获取文件列表
        md_list = os.listdir('./content/blog/%s/'%dir)
        # 逐个生成链接
        for md in md_list:
            filename = md.split('.',-1)[0]
            content = "> &emsp;&emsp;[%s](/blog/%s/%s)"%(filename, dir.lower(), filename.lower())
            f.write('%s\n> \n'%content)
            
# 打包
os.system('hugo')

# 更新简历
path_a = "C:\\Users\\Administrator\\hub\\missbless\\missbless\\resume"
path_b = "C:\\Users\\Administrator\\hub\\missbless\\missbless\\public\\resume"

os.system('robocopy %s %s /E'%(path_a,path_b))

# 配置代理
os.system('git config --global https.proxy https://127.0.0.1:1087')

# 推送代码

commit = input('commit log:')
os.system('cd public&&git add -A&&git commit -m "%s"&&git push -u origin master -f'%commit)

# 解除配置代理
os.system('git config --global --unset https.proxy')
