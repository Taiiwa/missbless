class QingDianActsPlugin(object):
    def __init__(self, gameId):
        # 实例化gameId
        self.gameId = gameId

    def event_handle(self, gameId):
        # 定义事件handle
        if gdata.serverType() == gdata.SRV_TYPE_UTIL:
            handles = {
                # 定义每个(cmd, action) 对应的函数名
                ('cmd，一般为game', 'action，自定义动作'): self.on_action,
                pvs.EVENT_CLICK_BTN_CHAT_REPLY: self.on_chat_replay,
                pvs.EVENT_COLLECT_RED_POINT: self.onCheckRedPoint,  # 红点
            }

            # 调试模式时增加的用于测试的一些handle
            if gdata.mode() != gdata.RUN_MODE_ONLINE:
                handles.update({
                    ("game", "debug_qdmemory_data"): self.on_qingdian_memory_user_data,
                })
            return handles

        return {}

    def on_action(self, gameId, msg):
        # handles
        # 获取参数，msg相当于request
        userId, cloudId = msg.getParams(u'userId', u'cloudId')
        # 获取数据并赋值
        # 获取ret、err_code、err_msg
        ret, err = plant.plant(userId)
        # 调用_common_send_msg，返回msg, ret, err_code, err_msg
        self._common_send_msg(msg, ret, err, '')

    def _common_send_msg(self, msg, ret, errCode, errInfo):
        """封装消息下发"""
        # 检查错误代码
        if errCode != const.ERR_OK:
            返回错误信息
            util.sendErrorMsg(errCode, errInfo)
            return
        # 根据request——msg声明一个相应数据response——mo
        mo = util.makeOutMsg(msg)
        # 如果没有错误，将传入的结果ret封装到mo中
        mo.updateResult(ret)
        # 将mo中的数据返回给前端
        util.routerOutMsg(mo)
